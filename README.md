# react-tab

### 安装

```shell
npm i react-tab -S
```

### 使用

```jsx
import { TabList, Tab } from 'react-tab';

class App extends React.Component {
  render() {
    return (
      <TabList>
        <Tab label="tab1">tab 1 content</Tab>
        <Tab label="tab2">tab 2 content</Tab>
        <Tab label="tab3">tab 3 content</Tab>
      </TabList>
    );
  }
}
```
